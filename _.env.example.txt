# ══════════════════════════════════════════════════════════════════
# ENVIRONMENT VARIABLES — copy to .env.local for local dev.
# On VERCEL: add each of these under Project → Settings → Environment
# Variables (set them for Production, Preview, and Development).
# ══════════════════════════════════════════════════════════════════

# ── Supabase (browser-side, safe to expose — read-only anon key) ──
# From Supabase → Project Settings → API
VITE_SUPABASE_URL=https://your-project-ref.supabase.co
VITE_SUPABASE_ANON_KEY=your_anon_key_here

# ── Groq AI (browser fallback only — server-side preferred) ───────
# The concierge and listing importer use /api/groq (server-side) first.
# VITE_GROQ_API_KEY is only a fallback if the function isn't reachable.
VITE_GROQ_API_KEY=gsk_your_groq_api_key_here

# ── PayHero / M-Pesa (SERVER-SIDE ONLY — no VITE_ prefix) ────────
# Read only by /api/payhero. From PayHero dashboard → API settings.
PAYHERO_USERNAME=your_payhero_username
PAYHERO_PASSWORD=your_payhero_password
PAYHERO_CHANNEL_ID=your_channel_id

# ── Groq AI (SERVER-SIDE — preferred) ────────────────────────────
# Powers the AI concierge AND the new "Paste details" listing import.
# Same key as VITE_GROQ_API_KEY, read server-side by /api/groq.
GROQ_API_KEY=gsk_your_groq_api_key_here

# ── Listing importer — residential-proxy scraper (OPTIONAL) ──────
# Airbnb & Booking.com block imports from datacenter IPs (Vercel included),
# so "Import from URL" alone can't fetch them. Set a scraping-API key to
# get past that block AND render their JavaScript (full photos + price).
# Works with ScraperAPI (default), Scrapingdog, ScrapingBee, ZenRows, etc.
# If you leave this unset, the importer still works for your own site and
# many OTAs, and always offers the "Paste details" AI fallback.
# SCRAPER_API_KEY=your_scraperapi_key
# SCRAPER_API_BASE=https://api.scraperapi.com/   # override for other providers
# SCRAPER_API_COUNTRY=ke                          # optional geo (ISO code)

# ── WhatsApp Business API (SERVER-SIDE) ───────────────────────────
# From Meta Business Manager → WhatsApp → API Setup.
WHATSAPP_TOKEN=your_meta_whatsapp_token
WHATSAPP_PHONE_NUMBER_ID=your_phone_number_id

# ── Google Places (SERVER-SIDE — for SOS emergency lookup) ────────
# Enable "Places API (New)" in Google Cloud + billing. Only needed if
# the SOS / emergency button is used. Read by /api/nearby-emergency.
GOOGLE_PLACES_API_KEY=your_google_places_api_key

# ── Supabase (SERVER-SIDE — for scheduled cron functions) ─────────
# Used by /api/auto-sync and /api/reservation-reminder, which run on
# a Vercel Cron schedule with no logged-in user.
# From Supabase → Project Settings → API → service_role key.
# WARNING: bypasses row-level security. Never expose it / never VITE_ it.
SUPABASE_URL=https://your-project-ref.supabase.co
SUPABASE_SERVICE_ROLE_KEY=your_service_role_key_here

# ── Site URL (SERVER-SIDE — used by reservation-reminder) ─────────
# Your deployed site's base URL, e.g. https://shikazhomez.com
# Vercel also exposes VERCEL_URL automatically; SITE_URL overrides it.
SITE_URL=https://your-domain.com

# ── Master developer password (SERVER-SIDE) ───────────────────────
# Lets you access any client's host portal. Read by /api/master-auth.
MASTER_PASSWORD=choose_a_strong_unique_password_per_client

# ── Web Push notifications (SERVER-SIDE — booking & sync alerts) ──
# Generate a keypair once with:  npx web-push generate-vapid-keys
# VAPID_PUBLIC_KEY must ALSO be exposed to the browser as VITE_VAPID_PUBLIC_KEY
# (same value). Without these, the app still works — push is just disabled.
VAPID_PUBLIC_KEY=your_vapid_public_key
VAPID_PRIVATE_KEY=your_vapid_private_key
VAPID_SUBJECT=mailto:admin@your-domain.com

# ── Web Push public key (BROWSER-SIDE — same value as VAPID_PUBLIC_KEY) ──
VITE_VAPID_PUBLIC_KEY=your_vapid_public_key
