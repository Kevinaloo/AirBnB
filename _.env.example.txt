# ══════════════════════════════════════════════════════════════════
# ENVIRONMENT VARIABLES — copy this file to .env.local for local dev
# All of these must be set in Netlify → Site Settings → Environment
# Variables for a deployed client site. See SETUP.md for full details.
# ══════════════════════════════════════════════════════════════════

# ── Supabase (browser-side, safe to expose — read-only anon key) ──
# From Supabase → Project Settings → API
VITE_SUPABASE_URL=https://your-project-ref.supabase.co
VITE_SUPABASE_ANON_KEY=your_anon_key_here

# ── Groq AI (browser fallback only — server-side preferred) ───────
# The concierge and listing importer use /api/groq (server-side) first.
# VITE_GROQ_API_KEY is only used as a fallback if the function isn't
# deployed yet. Ideally only set GROQ_API_KEY (server-side, below).
VITE_GROQ_API_KEY=gsk_your_groq_api_key_here

# ── PayHero / M-Pesa (SERVER-SIDE ONLY — no VITE_ prefix) ────────
# These MUST NOT have the VITE_ prefix. They stay off the browser
# bundle and are only read by netlify/functions/payhero.js.
# From your PayHero dashboard → API settings.
PAYHERO_USERNAME=your_payhero_username
PAYHERO_PASSWORD=your_payhero_password
PAYHERO_CHANNEL_ID=your_channel_id

# ── Groq AI (SERVER-SIDE — preferred over VITE_GROQ_API_KEY) ─────
# This powers the AI concierge and listing import refinement securely.
# Same key as VITE_GROQ_API_KEY above, but read server-side by
# netlify/functions/groq.js. Once this is set, VITE_GROQ_API_KEY is
# only used as a local dev fallback.
GROQ_API_KEY=gsk_your_groq_api_key_here

# ── WhatsApp Business API (SERVER-SIDE) ───────────────────────────
# From Meta Business Manager → WhatsApp → API Setup.
# See SETUP.md section 6 for full WhatsApp setup instructions.
WHATSAPP_TOKEN=your_meta_whatsapp_token
WHATSAPP_PHONE_NUMBER_ID=your_phone_number_id

# ── Google Places (SERVER-SIDE — for SOS emergency lookup) ────────
# From Google Cloud Console. Enable "Places API (New)" and add billing.
# Only required if the SOS / emergency button is being used.
GOOGLE_PLACES_API_KEY=your_google_places_api_key

# ── Supabase (SERVER-SIDE — for scheduled functions) ──────────────
# Used by netlify/functions/auto-sync.js and reservation-reminders.js
# which run on a schedule with no logged-in user.
# From Supabase → Project Settings → API → service_role key.
# WARNING: This key bypasses row-level security. Never expose it.
SUPABASE_URL=https://your-project-ref.supabase.co
SUPABASE_SERVICE_ROLE_KEY=your_service_role_key_here

# ── Master developer password (SERVER-SIDE) ───────────────────────
# Lets you access any client's host portal without knowing their PIN/
# password. Use something strong and unique per deployment.
# See netlify/functions/master-auth.js for details.
MASTER_PASSWORD=choose_a_strong_unique_password_per_client
